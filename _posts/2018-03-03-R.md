#Clasificador NaiveBayes
install.packages('e1071', dependencies=TRUE)

library(e1071)
train<-d[1:250,]
clasificador<-naiveBayes(marcaA ~ marcaB, data=train)

eval<-d[251:303,]
marcaA<-eval[0,6]
predicted<-predict(clasificador, eval[,-6])
matrizconf<-table(predicted, marcaA)
